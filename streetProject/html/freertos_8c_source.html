187void StartDefaultTask(void *argument) {
188 /* USER CODE BEGIN StartDefaultTask */
189 //Highest priority task which cycles through the different states
190 //Depending on the delay in Timer-task
191 //Set the GPIO to default, meaning no user input at initialisation
192 GPIO_State = Default;
193 /* Infinite loop */
194 for (;;) {
195 state = Vertical;
196 States(1); //Display the LEDS for when cars going vertically has the green light
197 vTaskDelay(1); // Small delay to prevent loss of data to the LEDS
198 vTaskSuspend(defaultTaskHandle);
199
200 state = Yellow;
201 States(3);
202 vTaskSuspend(defaultTaskHandle);
203
204 state = Horizontal;
205 States(2);
206 vTaskDelay(1);

208
